\section{Problemas}

\begin{proc}{ sudoku\_esTableroValido}{\In t: \matriz{\ent}, \Out result: \bool}{}{}
    \pre{\True}
    \post{result=esTableroValido(t)}
\end{proc}

\hfill

\begin{proc}{sudoku\_esCeldaVacia}{\In t: \matriz{\ent}, \In f: \ent,\In c: \ent, \Out result: \bool}{}{}
    \pre{esTableroValido(t) \yLuego 0 \leq f,c < |t|}
    \post{result=(t[f][c]=0)}
\end{proc}

\hfill

\begin{proc}{sudoku\_nroDeCeldasVacias}{\In t: \matriz{\ent}, \Out result: \ent}{}{}
    \pre{esTableroValido(t)}
    \post{\sum_{i=0}^{|t|-1} (\sum_{j=0}^{|t|-1} \IfThenElse{t[i][j] = 0 }{1}{0})}
\end{proc}

\hfill

\begin{proc}{sudoku\_primeraCeldaVaciaFila}{\In t: \matriz{\ent}, \Out result: \ent}{}{}
    \pre{esTableroValido(t)}
    \post{\IfThenElse{(\exists i: \ent)(0 \leq i < |t| \yLuego filaTieneCeldaVacia(t[i]) \yLuego (\forall j: \ent)\\    
    (0 \leq j < i \implicaLuego \neg filaTieneCeldaVacia(t[j])))\\}{result = i\\}{result = -1}}
\end{proc}

\hfill

\begin{proc}{sudoku\_primeraCeldaVaciaColumna}{\In t: \matriz{\ent}, \Out result: \ent}{}{}
	\pre{esTableroValido(t)}
	\post{\IfThenElse{(\exists i: \ent)(0 \leq i < |t| \yLuego filaTieneCeldaVacia(t[i]) \yLuego (\forall j: \ent)(0 \leq j < i \implicaLuego \neg filaTieneCeldaVacia(t[j])))\\}{result = indicePrimeraCeldaVaciaEnFila(t[i])\\}{result = -1}}
\end{proc}

\hfill

\begin{proc}{sudoku\_valorEnCelda}{\In t: \matriz{\ent}, \In f: \ent, \In c: \ent, \Out result: \ent}{}{}
	\pre{esTableroValido(t) \yLuego 0 \leq f,c \leq 8 \yLuego t[f][c] \neq 0}
	\post{result = t[f][c]}
\end{proc}

\hfill

\begin{proc}{sudoku\_llenarCelda}{\Inout t: \matriz{\ent}, \In f: \ent, \In c: \ent, \In value: \ent}{}{}
	\pre{esTableroValido(t) \yLuego 0 \leq f,c \leq 8 \yLuego t[f][c] = 0 \yLuego 1 \leq value \leq 9 \yLuego t=t_{0}}
	\post{t[f][c] = value \yLuego (\forall i: \ent)(\forall j: \ent)((0\leq i,j < |t| \yLuego (i \neq f \vee j \neq c)) \implicaLuego t[i][j]= t_{0}[i][j])}

\end{proc}

\hfill

\begin{proc}{sudoku\_vaciarCelda}{\Inout t: \matriz{\ent}, \In f: \ent, \In c: \ent}{}{}
	\pre{esTableroValido(t) \yLuego 0 \leq f,c \leq 8 \yLuego t[f][c] \neq 0 \yLuego t=t_{0}}
	\post{t[f][c] = 0 \yLuego (\forall i: \ent)(\forall j: \ent)((0 \leq i,j < |t| \yLuego (i \neq f \vee j \neq c)) \implicaLuego t[i][j] = t_{0}[i][j])}
\end{proc}

\hfill

\begin{proc}{sudoku\_esTableroParcialmenteResuelto}{\In t: \matriz{\ent}, \Out result: Bool)}{}{}
	\pre{\True}
	\post{result = esTableroParcialmenteResuelto(t)}
\end{proc}

\hfill

\begin{proc}{sudoku\_esTableroTotalmenteResuelto}{\In t: \matriz{\ent}, \Out result: \bool}{}{}

    \post{result = esTableroTotalmenteResuelto(t)}
    \end{proc}

\begin{proc}{sudoku\_esSubTablero}{\In t_{0},t_{1}: \matriz{\ent}, \Out result: \bool}{}{}
    \pre{\True}
    \post{result = esSubTablero(t_{0},t_{1})}
    \end{proc}
    
\hfill

\begin{proc}{sudoku\_tieneSolucion}{\In t: \matriz{\ent}, \Out tieneSolucion: \bool}{}{}
	\pre{esTableroValido(t)}
	\post{tieneSolucion = (\exists s: \matriz{\ent})(esTableroTotalmenteResuelto(s) \yLuego esSubTablero(s,t))}
\end{proc}

\hfill

\begin{proc}{sudoku\_resolver}{\Inout t: \matriz{\ent}, \Out tieneSolucion: \bool}{}{}
	\pre{esTableroValido(t) \yLuego t=t_{0}}
	\post{\IfThenElse{(\exists s: \matriz{\ent})(esTableroTotalmenteResuelto(s) \yLuego esSubTablero(s,t))\\}{tieneSolucion=\True \wedge t=s\\}{tieneSolucion = \False}}
\end{proc}

\hfill

\begin{proc}{sudoku\_copiarTablero}{\In src: \matriz{\ent}, \Out target: \matriz{\ent}}{}{}
	\pre{esTableroValido(src)}
	\post{esTableroValido(target) \yLuego (\forall i: \ent)(\forall j: \ent)(0 \leq i,j < |src| \implicaLuego target[i][j] = src[i][j])}
\end{proc}
   

\section{Predicados y Auxiliares generales}

    \pred{esMatriz}{t: \matriz{\ent}}{\\
    (\forall i: \ent)(\forall j: \ent)( 0\leq i,j <|t| \implicaLuego |t[i]| = |t[j]|)\\}

\hfill

    \pred{esMatrizCuadrada}{t: \matriz{\ent}}{\\
    esMatriz(t) \land (cantidadFilas(t) = cantidadColumnas(t))\\}

\hfill

    \pred{esTableroValido}{t: \matriz{\ent}}{   
        esMatrizCuadrada(t) \yLuego |t|=9 \yLuego\\
    (\forall i: \ent)(\forall j: \ent)( 0\leq i,j <|t| \implicaLuego 0 \leq t[i][j] \leq 9)\\}


\hfill

	\pred{filaTieneCeldaVacia}{f: \TLista{\ent}}{\\ 
	(\exists i: \ent)(0 \leq i < |f| \yLuego f[i] = 0)\\}

\hfill	

	\pred{noHayRepetidosEnRegion}{t: \matriz{\ent}}{
	(\forall i: \ent)(\forall j: \ent)(\forall k: \ent)(\forall l: \ent)\\
	(0\leq i,j,k,l < 9 \yLuego (i \text{ div } 3 = k \text{ div } 3) \yLuego (j \text{ div } 3 = l \text{ div } 3) \yLuego (i \neq k \vee j \neq l) \implicaLuego (t[i][j] = 0 \vee t[k][l] = 0 \vee t[i][j] \neq t[k][l]))\\}
	
\hfill

	\pred{noHayRepetidosEnFila}{s: \TLista{\ent}}{
	(\forall i: \ent)(\forall j: \ent)(0 \leq i,j < |s| \yLuego j \neq i \implicaLuego (s[i] = 0 \vee s[j] = 0 \vee s[i] \neq s[j]))\\}
	
\hfill

	\pred{noHayRepetidosEnColumna}{t: \matriz{\ent}}{
	(\forall j: \ent)(0 \leq j < |t| \implicaLuego (\forall l: \ent)(\forall k: \ent)\\
	(0 \leq l,k < |t| \yLuego l \neq k  \implicaLuego (t[l][j] = 0 \vee t[k][j] = 0 \vee t[l][j] \neq t[k][j])))\\}
	
\hfill

	\pred{esTableroParcialmenteResuelto}{t: \matriz{\ent}}{
	esTableroValido(t) \yLuego (\forall i: \ent)(0 \leq i < |t| \implicaLuego \\
	  noHayRepetidosEnFila(t[i])) \yLuego noHayRepetidosEnColumna(t) \yLuego noHayRepetidosEnRegion(t)\\}	
	
\hfill

	\pred{esTableroTotalmenteResuelto}{t: \matriz{\ent}}{
	esTableroParcialmenteResuelto(t) \yLuego (\forall i: \ent) (0 \leq i < |t| \implicaLuego \neg filaTieneCeldaVacia(t[i]))\\}	
	
\hfill

    \pred{esSubTablero}{t_{0},t_{1} : \matriz{\ent}}{
    esTableroValido(t_{0}) \yLuego  esTableroValido(t_{1})  \yLuego  (\forall i: \ent) (\forall j: \ent)\\    
    ( (0 \leq i |t| \yLuego t_{0}[i][j] \neq 0 ) \implicaLuego  t_{0}[i][j] = t_{1}[i][j])\\}	
	
	\aux{cantidadFilas}{t: \matriz{\ent}}{\ent}{|t|}
    
\hfill    
    
    \aux{cantidadColumnas}{t: \matriz{\ent}}{\ent}{
    \IfThenElse{cantidadFilas(t)>0}{|t[0]|}{0}} 

\hfill
