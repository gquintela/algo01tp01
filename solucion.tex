\begin{proc}{sudoku\_llenarCelda}{\Inout t: \matriz{\ent}, \In f: \ent, \In c: \ent, \In value: \ent}{}{}
	\pre{esTableroValido(t) \yLuego 0 \leq f,c \leq 8 \yLuego t[i][j] = 0 \yLuego 1 \leq value \leq 9 \yLuego t=t_{0}}
	\post{t[f][c] = value \yLuego (\forall i: \ent)(\forall j: \ent)(0\leq i,j \textless |t| \yLuego (i \neq f \vee j \neq c)) \implicaLuego t[i][j]= t_{0}[i][j]}

\end{proc}

\begin{proc}{sudoku\_vaciarCelda}{\Inout t: \matriz{\ent}, \In f: \ent, \In c: \ent}{}{}

\end{proc}

    esMatriz(t) $\land$ ( cantidadFilas(t) = cantidadColumnas(t) )}
